def generate_racking_data():
    racking_data = []

    aisles = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K']  # Aisles from A to K
    positions = range(1, 11)  # Positions from 01 to 10

    for aisle in aisles:
        for level in range(1, 6):  # Levels from 01 to 05
            for position in positions:
                racking_number = f"{aisle}-{str(level).zfill(2)}-{str(position).zfill(2)}"
                racking_data.append({"racking_number": racking_number, "aisle": aisle})

    return racking_data
